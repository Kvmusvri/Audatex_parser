                return request_data
            return None
        except Exception as e:
            logger.error(f"‚ùå –û—à–∏–±–∫–∞ –ø–æ–ª—É—á–µ–Ω–∏—è –∑–∞—è–≤–∫–∏ –∏–∑ –æ—á–µ—Ä–µ–¥–∏: {e}")
